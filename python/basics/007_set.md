# üß© 007: Managing Unique Items with Sets

Sets are Python's unordered, unique collections‚Äîperfect for deduplication, fast membership tests, and mathematical set logic. Use curly braces or set() to create them. Sets are mutable, but their elements must be immutable. They're ideal for filtering, combining, and comparing data, and are much faster than lists for membership checks.

## üéØ Python Set: Practical, Tricky, and Fun Usages

```python
def set_usages():
    # ===== 1. Basic Creation & Membership =====
    s = {1, 2, 3}
    s2 = set([4, 5, 6, 6])
    print(s, s2)
    print(2 in s, 7 in s2)

    # ===== 2. Set Comprehension Magic =====
    squares = {x**2 for x in range(6)}
    evens = {x for x in range(10) if x % 2 == 0}
    print(squares, evens)

    # ===== 3. Adding, Removing, Modifying =====
    s.add(4)
    s.discard(2)
    s.update([5, 6])
    s.remove(1)
    print(s)

    # ===== 4. Mathematical Set Operations =====
    a = {1, 2, 3}
    b = {3, 4, 5}
    print(a | b, a & b, a - b, a ^ b)

    # ===== 5. Conversion Tricks =====
    lst = [1, 2, 2, 3]
    unique = set(lst)
    t = tuple(unique)
    l = list(unique)
    print(unique, t, l)

    # ===== 6. Boolean & Truthiness =====
    print(bool(set()), bool({1}))

    # ===== 7. Frozen Sets =====
    fs = frozenset([1, 2, 3])
    print(fs)

    # ===== 8. Common Pitfalls =====
    # Mutable elements not allowed
    # s = {[1,2], 3}  # ‚ùå TypeError
    empty = set()
    not_a_set = {}
    print(type(empty), type(not_a_set))

    # ===== 9. Performance Considerations =====
    import time
    big_list = list(range(10**6))
    big_set = set(big_list)
    start = time.time()
    999999 in big_list
    print("List membership time:", time.time() - start)
    start = time.time()
    999999 in big_set
    print("Set membership time:", time.time() - start)

    # ===== 10. Fun & Advanced Tricks =====
    s = "banana"
    print(set(s))
    print(set("hello") & set("world"))
    # Set union of multiple sets
    print(set.union({1,2}, {2,3}, {3,4}))
    # Set intersection of multiple sets
    print(set.intersection({1,2,3}, {2,3,4}, {3,4,5}))
    # Remove duplicates from a list
    data = [1,2,2,3,3,4]
    print(list(set(data)))

set_usages()
```
